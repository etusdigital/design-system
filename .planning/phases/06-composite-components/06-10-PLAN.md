```typescript
import { useState } from 'react';
import type { Meta, StoryObj } from '@storybook/react';
import { ComponentName } from './ComponentName';

const meta = {
  component: ComponentName,
  argTypes: { /* prop descriptions */ },
} satisfies Meta<typeof ComponentName>;

export default meta;
type Story = StoryObj<typeof ComponentName>;

export const Primary: Story = {
  render: () => {
    const [value, setValue] = useState(defaultValue);
    return <ComponentName value={value} onChange={setValue} />;
  },
};
```

Key conversion rules:
- `@storybook/vue3` → `@storybook/react`
- `import Foo from './Foo.vue'` → `import { Foo } from './Foo'`
- Vue `v-model="x"` → React `value={x} onChange={handler}`
- Vue `v-model:expanded="x"` → React `expanded={x} onExpandedChange={handler}`
- Vue `:prop="val"` → React `prop={val}`
- Vue `@click="fn"` → React `onClick={fn}`
- Vue `{{ expr }}` → React `{expr}`
- Vue template slots → React `children` prop
- Vue `<template #header>` → React `header={<>...</>}` prop
- `modelValue` prop → `value` prop (React component uses `value`, not `modelValue`)
- Each story render function that uses state must use `useState` inside the render function
- No `defaultRender` helper pattern needed — each story is self-contained
