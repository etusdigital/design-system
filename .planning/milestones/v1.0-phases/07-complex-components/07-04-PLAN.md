**Client-side sorting (port from Table.vue):**
```typescript
function sortItems(items: any[], sortKey: string, desc: boolean): any[] {
  return [...items].sort((a, b) => {
    const aVal = a[sortKey];
    const bVal = b[sortKey];
    if (aVal < bVal) return desc ? 1 : -1;
    if (aVal > bVal) return desc ? -1 : 1;
    return 0;
  });
}
```

**Sort handler:**
```typescript
const handleSort = (column: Column) => {
  if (!column.sortable) return;
  const key = column.value;
  const newDesc = sortByName === key ? !isDesc[key] : false;
  setSortByName(key);
  setIsDesc(prev => ({ ...prev, [key]: newDesc }));
  onSortBy?.(key, newDesc);
};
```

**Selection (Set-based, NOT mutating items):**
```typescript
const handleSelectRow = (index: number, item: any) => {
  setSelectedRows(prev => {
    const next = new Set(prev);
    const wasSelected = next.has(index);
    wasSelected ? next.delete(index) : next.add(index);
    onSelectRow?.(item, !wasSelected);
    return next;
  });
};

const handleSelectAll = (selectAll: boolean) => {
  if (selectAll) {
    setSelectedRows(new Set(pagedItems.map((_, i) => i)));
  } else {
    setSelectedRows(new Set());
  }
  setAllSelected(selectAll);
  onSelectAll?.(selectAll);
};
```

            </th>

        </tr>
      </thead>
      <tbody>
        {loading ? (

            <tr key={rowIndex} className={clsx(hasHover && styles.rowHover, selectedRows.has(rowIndex) && styles.rowSelected)}>

              {enableAggregation && <td>{((page ?? 1) - 1) * (itemsPerPage ?? 10) + rowIndex + 1}</td>}
              {columns.map(col => (
                <td key={col.value} style={{ textAlign: col.align }}>
                  {col.render ? col.render(item[col.value], item, rowIndex) : item[col.value]}
                </td>
              ))}
              {actionsSlot && <td>{/* actions rendered per row */}</td>}
            </tr>
          ))
        )}
      </tbody>
    </table>
