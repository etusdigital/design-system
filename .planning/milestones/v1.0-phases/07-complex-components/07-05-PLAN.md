2. **Selection save/restore (CRITICAL for toolbar):**
```typescript
const savedSelectionRef = useRef<Range | null>(null);

function saveCurrentSelection() {
  const selection = window.getSelection();
  if (!selection || selection.rangeCount === 0) return;
  const range = selection.getRangeAt(0);
  if (editorRef.current?.contains(range.commonAncestorContainer))
    savedSelectionRef.current = range.cloneRange();
}

function restoreSavedSelection() {
  const selection = window.getSelection();
  if (!savedSelectionRef.current || !editorRef.current || !selection) return;
  editorRef.current.focus();
  selection.removeAllRanges();
  const range = savedSelectionRef.current;
  if (editorRef.current.contains(range.commonAncestorContainer))
    selection.addRange(range);
}
```

3. **History (undo/redo) with isRestoringHistory ref:**
```typescript
const historyRef = useRef<string[]>([]);
const historyIndexRef = useRef(-1);
const isRestoringHistoryRef = useRef(false);

function saveToHistory(html: string) {
  const history = historyRef.current;
  if (historyIndexRef.current < history.length - 1) {
    history.splice(historyIndexRef.current + 1);
  }
  history.push(html);
  historyIndexRef.current = history.length - 1;
}

function undoAction() {
  if (historyIndexRef.current <= 0 || !editorRef.current) return;
  isRestoringHistoryRef.current = true;
  historyIndexRef.current--;
  editorRef.current.innerHTML = historyRef.current[historyIndexRef.current];
  isRestoringHistoryRef.current = false;
  const html = editorRef.current.innerHTML;
  setValue(html);
}

function redoAction() {
  if (historyIndexRef.current >= historyRef.current.length - 1 || !editorRef.current) return;
  isRestoringHistoryRef.current = true;
  historyIndexRef.current++;
  editorRef.current.innerHTML = historyRef.current[historyIndexRef.current];
  isRestoringHistoryRef.current = false;
  const html = editorRef.current.innerHTML;
  setValue(html);
}
```
