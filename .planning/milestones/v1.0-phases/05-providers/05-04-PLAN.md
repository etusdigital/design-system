export interface DialogProps {
  value?: boolean;
  onChange?: (value: boolean) => void;
  width?: string;
  height?: string;
  noOutsideClose?: boolean;
  children?: React.ReactNode;
  className?: string;
}
export function Dialog(props: DialogProps): JSX.Element;

<!-- Toast provider API (from Toast.tsx) -->
```typescript
export interface ToastOptions {
  id?: string;
  title?: string;
  message?: string;
  type?: 'info' | 'success' | 'warning' | 'danger' | 'neutral';
  top?: boolean;
  bottom?: boolean;
  left?: boolean;
  right?: boolean;
  timeout?: number | null;
  buttonLabel?: string;
  action?: () => void;
}
export function ToastProvider({ children }: { children: React.ReactNode }): JSX.Element;
export function useToast(): { toast: (options: ToastOptions) => { id: string; close: () => void } };
```

<!-- Confirm provider API (from Confirm.tsx) -->
```typescript
interface ConfirmOptions {
  title?: string;
  message?: string;
  acceptLabel?: string;
  cancelLabel?: string;
}
export function ConfirmProvider({ children }: { children: React.ReactNode }): JSX.Element;
export function useConfirm(): { confirm: (options: ConfirmOptions) => Promise<boolean> };
```
